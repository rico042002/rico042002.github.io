Temperature Conversion

JavaScript Code will prompt the user for temperature in degrees Celsius and convert to Fahrenheit

function convertToF(celsius) {
  let fahrenheit;
 
  let fahrenheit = (celsius * (9/5)) + 32;
  
  return fahrenheit;
}

convertToF(45);

FizzBuzz

Solution to the dreaded Fizz Buzz algorithm


function fizzBuzz() {
for (var i = 0; i <= 100; i++) {
switch (true) {
	case i % 3 === 0 && i % 5 === 0:
		console.log("FizzBuzz");
		break;
	case i % 3 === 0:
		console.log("Fizz");
	break;
	case i % 5 === 0:
		console.log("Buzz");
		}
	}
}
fizzBuzz();

What is Recursion?

Example of a simple Recursive Function


/* Recursion is a function that calls itself until 
it no longer doesn't */

let countDownFrom = (num) => {
    if (num === 0) return;
	console.log(num);
	countDownFrom(num - 1);
};

countDownFrom(10);
